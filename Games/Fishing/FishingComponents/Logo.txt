func Reset()
  isDisabled = true
  logoIn = false
  logoOut = false


func Show()
  isDisabled = false
  logoIn = true
  logoOut = false
  panel.x = 0


func Hide()
  isDisabled = false
  logoIn = false
  logoOut = true


func Update()
  ?isDisabled
    return

  ?logoOut
    panel.x += 2
    ?panel.x >= 0
      Reset()
  :?logoIn
    panel.x -= 2
    ?panel.x <= -44
      panel.x = -44
      logoIn = false
